# @cfware/button [![NPM Version][npm-image]][npm-url]

Icon button component

## Usage

```js
import {html} from '@cfware/shadow-element';
import '@cfware/button';

const buttonHTML = () => html`
	<cfware-button icon="\uF0F2" />
`;
```
